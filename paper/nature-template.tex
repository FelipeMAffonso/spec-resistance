% Nature-style LaTeX template for Pandoc
% Produces single-column, publication-quality output matching Nature/NMI style
\documentclass[11pt]{article}

% ── Page geometry ──────────────────────────────────────────────
\usepackage[
  paperwidth=210mm,
  paperheight=297mm,
  left=20mm,
  right=20mm,
  top=25mm,
  bottom=25mm
]{geometry}

% ── Fonts ──────────────────────────────────────────────────────
\usepackage{fontspec}
\setmainfont{Palatino Linotype}[
  BoldFont=Palatino Linotype Bold,
  ItalicFont=Palatino Linotype Italic,
  BoldItalicFont=Palatino Linotype Bold Italic
]
\setsansfont{Arial}
\setmonofont{Consolas}

% ── Spacing & typography ──────────────────────────────────────
\usepackage{setspace}
\setstretch{1.15}
\setlength{\parskip}{0.3ex plus 0.1ex minus 0.1ex}
\setlength{\parindent}{1.2em}

% ── Colours ────────────────────────────────────────────────────
\usepackage{xcolor}
\definecolor{nmiblue}{RGB}{0,109,119}
\definecolor{headinggray}{RGB}{80,80,80}
\definecolor{naturegrey}{RGB}{100,100,100}

% ── Section headings (Nature style, gray serif) ──────────────
\usepackage{titlesec}
\setcounter{secnumdepth}{0}  % No section numbering (Nature style)

\titleformat{\section}
  {\Large\bfseries}
  {}
  {0pt}
  {}
\titlespacing*{\section}{0pt}{3ex plus 1ex minus 0.5ex}{1.2ex plus 0.3ex}

\titleformat{\subsection}
  {\large\bfseries}
  {}
  {0pt}
  {}
\titlespacing*{\subsection}{0pt}{2.5ex plus 0.8ex minus 0.3ex}{0.8ex plus 0.2ex}

\titleformat{\subsubsection}
  {\normalsize\bfseries}
  {}
  {0pt}
  {}
\titlespacing*{\subsubsection}{0pt}{2ex plus 0.5ex minus 0.2ex}{0.5ex plus 0.1ex}

% ── Figures & tables ──────────────────────────────────────────
\usepackage{graphicx}
\usepackage{float}
\floatplacement{figure}{htbp}
\setlength{\intextsep}{3ex plus 1ex minus 0.5ex}
\setlength{\floatsep}{4ex plus 1.5ex minus 0.5ex}
\setlength{\textfloatsep}{3ex plus 1ex minus 0.5ex}
\setcounter{topnumber}{1}
\setcounter{bottomnumber}{1}
\setcounter{totalnumber}{1}
\usepackage{caption}
% Define pipe separator manually (Nature uses "Figure 1 | Caption text")
\DeclareCaptionLabelSeparator{naturepipe}{\enspace\textbar\enspace}
\captionsetup{
  font={footnotesize},
  labelformat=empty,
  justification=justified,
  singlelinecheck=false,
  skip=6pt
}

% ── Tables ────────────────────────────────────────────────────
\usepackage{booktabs}
\usepackage{longtable}
\usepackage{array}
\usepackage{calc}    % Required for Pandoc \real{} column width calculations

% Shrink font size for all longtables to help wide tables fit
\usepackage{etoolbox}
\AtBeginEnvironment{longtable}{\footnotesize}
\AtBeginEnvironment{figure}{\setlength{\parindent}{0pt}}

% ── Lists ─────────────────────────────────────────────────────
\usepackage{enumitem}
\setlist{nosep, leftmargin=1.5em}

% ── Links ─────────────────────────────────────────────────────
\usepackage{hyperref}
\usepackage{xurl}
\urlstyle{same}
\hypersetup{
  colorlinks=true,
  linkcolor=nmiblue,
  citecolor=nmiblue,
  urlcolor=nmiblue,
  pdfborder={0 0 0}
}

% ── Header rule ───────────────────────────────────────────────
\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhf{}
\renewcommand{\headrulewidth}{0pt}
\fancyfoot[C]{\small\thepage}

% ── Line numbers (Nature submission requirement) ──────────────
\usepackage{lineno}
\linenumbers
\renewcommand\linenumberfont{\normalfont\tiny\color{naturegrey}}

% ── Pandoc tightlist ──────────────────────────────────────────
\providecommand{\tightlist}{%
  \setlength{\itemsep}{0pt}\setlength{\parskip}{0pt}}

\setlength{\emergencystretch}{3em}

% ── CSL references (Pandoc 3.x citeproc) ────────────────────
\newcommand{\citeproctext}{}  % Required by Pandoc 3.x citeproc
\newcommand{\citeproc}[2]{\begingroup\def\citeproctext{#2}\hyperlink{#1}{\citeproctext}\endgroup}
\newlength{\cslhangindent}
\setlength{\cslhangindent}{1.5em}
\newlength{\csllabelwidth}
\setlength{\csllabelwidth}{3em}
\newenvironment{CSLReferences}[2]
 {\begin{list}{}%
  {\setlength{\itemindent}{0pt}
   \setlength{\leftmargin}{0pt}
   \setlength{\parsep}{0pt}
   \setlength{\itemsep}{0.15em}
   \setlength{\topsep}{0pt}
   \ifnum#1=1
    \setlength{\leftmargin}{\cslhangindent}
    \setlength{\itemindent}{-\cslhangindent}
   \fi}
  \small
  % Redefine \bibitem inside CSLReferences to suppress [] brackets
  \def\bibitem[##1]##2{\item\leavevmode\hypertarget{##2}{}}}
 {\end{list}}
\newcommand{\CSLBlock}[1]{#1\hfill\break}
\newcommand{\CSLLeftMargin}[1]{\parbox[t]{\csllabelwidth}{#1}}
\newcommand{\CSLRightInline}[1]{\parbox[t]{\dimexpr\linewidth-\csllabelwidth}{#1}\break}
\newcommand{\CSLIndent}[1]{\hspace{\cslhangindent}#1}
